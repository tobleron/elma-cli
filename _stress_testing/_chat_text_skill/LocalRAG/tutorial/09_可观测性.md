# 第九章：可观测性（Observability）

## 什么是可观测性？

可观测性 = 你能看到系统运行时的内部状态。对于 RAG 系统，关键问题是：

- **LLM 查询是否成功？失败了怎么办？**
- **Embedding 延迟多少？缓存命中率多少？**
- **向量检索找到了多少相关文档？**
- **服务崩溃了怎么看原因？**

三个核心工具：**日志（Logs）、指标（Metrics）、追踪（Traces）**。

## 9.1 结构化日志（structlog）

普通日志是字符串，grep 搜索很困难：

```python
# 普通日志（不好）
logger.info("Embedding completed in 0.23 seconds")
logger.info("Retrieved 5 documents")
logger.info("User feedback: thumbs_up")
# 问题：无法用程序解析，不知道每条日志属于哪个请求
```

结构化日志把每条日志变成**键值对**：

```python
# 结构化日志（好）
logger.info(
    "embedder.batch_completed",
    duration_seconds=0.23,
    batch_size=16,
    cache_hit_rate=0.75,
    request_id="req-abc123",
)
# → JSON: {"event": "embedder.batch_completed", "duration_seconds": 0.23, ...}
# 可以用工具分析：查某请求的所有日志、按 duration 过滤、按字段聚合...
```

```python
# src/observability.py

import structlog
import logging
import sys

def configure_logging(log_file: str = "logs/app.log", log_level: str = "INFO"):
    """
    配置结构化日志。
    输出到：1. 控制台（彩色） 2. 文件（JSON 格式，轮转）
    """
    # 1. 配置标准 logging
    logging.basicConfig(
        format="%(message)s",
        level=getattr(logging, log_level.upper()),
        handlers=[logging.StreamHandler(sys.stdout)],
    )

    # 2. 配置 structlog
    structlog.configure(
        processors=[
            structlog.contextvars.merge_contextvars,      # 合并请求上下文
            structlog.processors.add_log_level,          # 添加日志级别
            structlog.processors.TimeStamper(fmt="iso"), # ISO 时间戳
            structlog.processors.StackInfoRenderer(),    # 异常堆栈
            structlog.processors.format_exc_info,        # 格式化异常
            # 控制台：彩色输出
            structlog.dev.ConsoleRenderer() if sys.stdout.isatty() else
            # 文件：JSON 输出
            structlog.processors.JSONRenderer(),
        ],
        wrapper_class=structlog.make_filtering_bound_logger(
            getattr(logging, log_level.upper())
        ),
        logger_factory=structlog.PrintLoggerFactory(),
        cache_logger_on_first_use=True,
    )

    # 3. 添加文件 Handler（带轮转）
    import logging.handlers
    file_handler = logging.handlers.RotatingFileHandler(
        log_file,
        maxBytes=10 * 1024 * 1024,  # 10MB
        backupCount=5,               # 保留 5 个备份
    )
    file_handler.setFormatter(logging.Formatter("%(message)s"))
    logging.getLogger().addHandler(file_handler)

    return structlog.get_logger()
```

**日志轮转**（RotatingFileHandler）的作用：
```
app.log        ← 当前日志文件（10MB 写满后）
app.log.1      ← 最近一次轮转的备份
app.log.2      ← 第二次轮转的备份
app.log.3
app.log.4
app.log.5      ← 最老的备份，超过 5 个就删除
```

## 9.2 Prometheus 指标

Prometheus 是一个时序数据库，专门存"数字随时间变化"的指标。

### 三种指标类型

| 类型 | 用途 | RAG 示例 |
|------|------|---------|
| **Counter** | 只增不减（累计数） | `rag_queries_total`（总查询次数），`api_errors_total` |
| **Gauge** | 可增可减（当前值） | `embedder_cache_size`（缓存当前大小），`indexed_docs_count` |
| **Histogram** | 统计分布（带桶） | `rag_query_duration_seconds`（延迟分布），`api_latency_seconds` |

### 代码实现

```python
# src/observability.py

from prometheus_client import Counter, Gauge, Histogram, start_http_server

# 1. 定义指标
RAG_QUERIES_TOTAL = Counter(
    "rag_queries_total",
    "Total RAG queries",
    ["status"]  # 标签：success / failure
)

RAG_QUERY_DURATION = Histogram(
    "rag_query_duration_seconds",
    "RAG query latency",
    buckets=[0.1, 0.25, 0.5, 1.0, 2.5, 5.0, 10.0],
)

API_LATENCY = Histogram(
    "api_call_duration_seconds",
    "External API call latency",
    ["client", "operation"],  # client=embedder/llm, operation=embed_single/batch/generate
)

VECTORSTORE_OPERATIONS = Histogram(
    "vectorstore_operation_duration_seconds",
    "VectorStore operation latency",
    ["operation"],  # operation=query/add/delete
)

EMBEDDER_CACHE_HITS = Counter(
    "embedder_cache_hits_total",
    "Embedder cache hits",
)

INDEXED_DOCS_COUNT = Gauge(
    "indexed_documents_count",
    "Number of indexed documents",
)

INDEXED_CHUNKS_COUNT = Gauge(
    "indexed_chunks_count",
    "Number of indexed chunks",
)

# 2. 启动指标 HTTP 服务器（后台线程）
def start_metrics_server(port: int = 9090):
    """启动 Prometheus 指标端点（不阻塞主线程）。"""
    from threading import Thread
    thread = Thread(target=start_http_server, args=(port,), daemon=True)
    thread.start()
    print(f"Prometheus metrics server running on http://localhost:{port}/metrics")
```

### 在业务代码中使用

```python
# 在 pipeline.py 的 query() 方法中

import time
from observability import (
    RAG_QUERIES_TOTAL, RAG_QUERY_DURATION,
    API_LATENCY, VECTORSTORE_OPERATIONS,
)

def query(self, question: str, ...):
    start = time.perf_counter()

    try:
        # 检索
        t0 = time.perf_counter()
        docs = self.retriever.retrieve(question)
        VECTORSTORE_OPERATIONS.labels(operation="retrieve").observe(time.perf_counter() - t0)

        # LLM 生成
        t0 = time.perf_counter()
        answer = self.llm.generate(prompt)
        API_LATENCY.labels(client="llm", operation="generate").observe(time.perf_counter() - t0)

        # 记录成功
        RAG_QUERIES_TOTAL.labels(status="success").inc()
        return {"answer": answer, "sources": docs}

    except Exception as e:
        RAG_QUERIES_TOTAL.labels(status="failure").inc()
        raise

    finally:
        RAG_QUERY_DURATION.observe(time.perf_counter() - start)
```

## 9.3 请求追踪（Context）

日志分散在各个函数中，怎么知道"这条日志属于那个请求"？

```python
from contextvars import ContextVar

# 全局 context variable（请求级别）
request_context: ContextVar[dict] = ContextVar("request_context", default={})

def log_with_context(logger, event: str, **kwargs):
    """把 context（request_id、user_id 等）自动注入日志。"""
    ctx = request_context.get()
    logger.info(event, **kwargs, **ctx)

# ---- 中间件/入口点 ----
def rag_endpoint(question: str):
    # 每个请求分配一个 ID
    request_id = str(uuid.uuid4())[:8]

    # 把 request_id 注入到 context
    token = request_context.set({"request_id": request_id})
    try:
        result = pipeline.query(question)
        log_with_context(logger, "query_completed",
            question_length=len(question),
            answer_length=len(result["answer"]),
        )
        return result
    finally:
        request_context.reset(token)  # 请求结束后清理
```

所有在这个请求中打的日志，都会自动带上 `"request_id": "abc12345"`，方便排查。

## 9.4 完整可观测性配置

```python
# src/app.py 启动时

from observability import configure_logging, start_metrics_server, get_logger

# 1. 配置日志
configure_logging(log_file="logs/app.log", log_level="INFO")
logger = get_logger(__name__)

# 2. 启动指标服务器（后台线程）
start_metrics_server(port=9090)

# 3. 在关键路径记录日志
logger.info("app_started", port=8501, version="0.3.0")
```

## 可观测性建设清单

| 维度 | 实现 | 用途 |
|------|------|------|
| 日志 | structlog + RotatingFileHandler | 查 Bug、追踪请求 |
| 指标 | Prometheus Counter/Histogram/Gauge | 看趋势、设告警 |
| 追踪 | ContextVar request_id | 关联同一请求的所有日志 |
| 健康检查 | `/health` 端点 | K8s/负载均衡探测 |
