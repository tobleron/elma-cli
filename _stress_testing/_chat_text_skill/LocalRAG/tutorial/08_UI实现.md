# 第八章：UI 实现（Streamlit）

## 8.1 Streamlit 简介

Streamlit 是一个 Python 库，可以快速构建 Web UI，完全不需要 HTML/CSS/JavaScript：

```python
import streamlit as st

st.title("我的 RAG 应用")
st.write("你好，世界！")
```

运行 `streamlit run app.py`，浏览器打开 `http://localhost:8501`，一个 Web 应用就完成了。

## 8.3 session_state（状态持久化）

Streamlit 默认每次交互都重新执行整个脚本，变量会丢失。`st.session_state` 是 Python 字典，可以跨请求保持数据：

```python
# 每次重新执行，messages 不会丢失
if "messages" not in st.session_state:
    st.session_state.messages = []

st.session_state.messages.append({"role": "user", "content": question})
```

def render_rag_tab():

            st.session_state.messages = []

        # 列出所有历史对话
        history = history_manager.list_conversations()
        for conv in history:
            col1, col2 = st.columns([5, 1])
            with col1:
                if st.button(
                    conv["title"][:20],
                    key=f"load_{conv['id']}",
                    use_container_width=True,
                ):
                    st.session_state.conversation_id = conv["id"]
                    st.session_state.messages = history_manager.get_conversation(conv["id"])
                    st.rerun()
            with col2:
                if st.button("🗑", key=f"del_{conv['id']}"):
                    history_manager.delete_conversation(conv["id"])
                    st.rerun()

        # 反馈统计
        stats = history_manager.get_feedback_stats()
        st.divider()
        st.caption(f"👍 {stats['thumbs_up']}  👎 {stats['thumbs_down']}")

    # ---- 右侧：聊天主区域 ----
    st.title("📚 RAG 问答")

    # 1. 渲染历史消息
    for msg in st.session_state.messages:
        if msg["role"] == "user":
            st.chat_message("user").write(msg["content"])
        else:
            with st.chat_message("assistant"):
                st.write(msg["content"])
                # 反馈按钮
                render_feedback_buttons(msg)

    # 2. 用户输入
    if question := st.chat_input("输入你的问题..."):
        # 保存用户消息
        st.session_state.messages.append({"role": "user", "content": question})
        st.chat_message("user").write(question)

        # 生成回答
        with st.chat_message("assistant"):
            with st.spinner("正在检索并生成回答..."):
                result = rag_pipeline.query(question)

            st.write(result["answer"])

            # 显示来源
            if result["sources"]:
                with st.expander("📄 参考来源"):
                    for i, src in enumerate(result["sources"], 1):
                        st.markdown(f"**[{i}]** {src['source']}")
                        st.caption(src["text"][:200] + "...")

            # 保存到历史
            st.session_state.messages.append({
                "role": "assistant",
                "content": result["answer"],
                "sources": result["sources"],
            })

            # 反馈按钮
            render_feedback_buttons(result["answer"])

## 8.5 反馈按钮实现

用户可以对每个回答点 thumbs up/down，数据保存到 JSON：

```python
def render_feedback_buttons(message_or_text):
    """在回答下方渲染反馈按钮。"""

    # 用内容的哈希作为按钮 key（唯一且稳定）
    msg_hash = hashlib.sha256(message_or_text.encode()).hexdigest()[:8]

    col_up, col_down = st.columns([1, 1])
    with col_up:
        if st.button("👍", key=f"up_{msg_hash}"):
            history_manager.save_feedback(
                question=st.session_state.messages[-2]["content"],  # 用户的问题
                answer=message_or_text,
                rating="thumbs_up",
            )
            st.success("已记录反馈，感谢！")
            st.rerun()

    with col_down:
        if st.button("👎", key=f"down_{msg_hash}"):
            history_manager.save_feedback(
                question=st.session_state.messages[-2]["content"],
                answer=message_or_text,
                rating="thumbs_down",
            )
            st.success("已记录反馈，感谢！")
            st.rerun()
```

            if st.button("🗑 删除", key=f"del_doc_{doc['source']}"):
                vectorstore.delete_document(doc["source"])
                st.rerun()

    # 清空全部
    st.divider()
    if st.button("⚠️ 清空所有文档", type="danger"):
        vectorstore.delete_all()

## 8.8 Streamlit 架构总结

| 概念 | 说明 |
|------|------|
| `st.session_state` | 全局状态字典，跨请求保持数据 |
| `st.tabs()` | 多标签页切换 |
| `st.chat_message()` | 聊天消息 UI |
| `st.expander()` | 可折叠内容区 |
| `st.form()` + `st.form_submit_button()` | 表单（批量提交） |
| `st.rerun()` | 重新执行脚本（刷新 UI） |
